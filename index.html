// --- ตั้งค่าตัวแปร (อย่าลืมเปลี่ยนเป็น Key ใหม่เพื่อความปลอดภัยนะครับ) --- const LINE_ACCESS_TOKEN = 'ใส่_LINE_ACCESS_TOKEN_ใหม่ที่นี่'; const GEMINI_API_KEY = 'ใส่_GEMINI_API_KEY_ใหม่ที่นี่'; const SHEET_ID = '17DOifrj6Tff_6QFyFURV5dQ4PsR5xq5rOgWg6GmcUNI'; const SHEET_NAME = 'Sheet1'; // --- 2. ตั้งค่าสถานที่เช็คชื่อ (พิกัดและระยะ) --- const TARGET_LAT = 19.5564; const TARGET_LNG = 99.7373; const ALLOWED_RADIUS = 100; // ระยะห่างที่อนุญาต (หน่วย: เมตร) // ========================================== // ส่วนรับข้อมูล Webhook จาก LINE & LIFF Fetch // ========================================== function doPost(e) { if (!e || !e.postData || !e.postData.contents) return ContentService.createTextOutput("OK"); const data = JSON.parse(e.postData.contents); const now = new Date(); const dateStr = Utilities.formatDate(now, "GMT+7", "dd/MM/yyyy"); // === A. รับข้อมูลจากหน้าเว็บ LIFF === if (data.action === 'liff_checkin') { try { // 1. เช็คว่าเป็นการ "มาเรียน" และมีการส่งพิกัดมาหรือไม่ if (data.status === 'มาเรียน') { if (!data.latitude || !data.longitude) { return sendJsonResponse(false, "ไม่พบพิกัด GPS กรุณาเปิดระบบ Location"); } // คำนวณระยะทาง const distance = calculateDistance(data.latitude, data.longitude, TARGET_LAT, TARGET_LNG); if (distance > ALLOWED_RADIUS) { return sendJsonResponse(false, `คุณอยู่นอกพื้นที่ (${Math.round(distance)} เมตรจากจุดเช็คชื่อ)`); } } // 2. เช็คว่าวันนี้เคยบันทึกข้อมูลไปแล้วหรือยัง if (hasCheckedInToday(data.userId, dateStr)) { return sendJsonResponse(false, "วันนี้คุณได้ทำการเช็คชื่อไปแล้วครับ"); } // 3. บันทึกข้อมูลหากผ่านเงื่อนไขทั้งหมด saveDataToSheet(data.userId, data.displayName, data.status, 'LIFF'); return sendJsonResponse(true, `บันทึกสถานะ "${data.status}" เรียบร้อยแล้ว`); } catch (error) { return sendJsonResponse(false, "System Error: " + error.toString()); } } // === B. รับข้อความแชทจาก LINE === const event = data.events[0]; if (event && event.type === 'message' && event.message.type === 'text') { const text = event.message.text.trim(); const userId = event.source.userId; const replyToken = event.replyToken; if (text === 'มาเรียน') { replyText(replyToken, "📍 เพื่อความถูกต้อง กรุณากดปุ่มเช็คชื่อผ่าน menus (LIFF) เพื่อยืนยันพิกัดสถานที่ครับ"); } else if (text === 'ลา' || text === 'ขาด') { const profile = getUserProfile(userId); const displayName = profile.displayName || "ไม่ทราบชื่อ"; if (hasCheckedInToday(userId, dateStr)) { replyText(replyToken, "⚠️ วันนี้คุณได้ทำการบันทึกข้อมูลไปแล้วครับ"); } else { saveDataToSheet(userId, displayName, text, 'Chat'); replyText(replyToken, `✅ บันทึกสถานะ "${text}" เรียบร้อยครับ`); } } else if (/สรุป|ยอด|ประวัติ|กี่ครั้ง|ขาดเรียน|เช็คชื่อ|รายงาน/i.test(text)) { const profile = getUserProfile(userId); const attendanceSummary = getUserAttendanceSummary(userId); const geminiResponse = callGeminiWithAttendance(text, profile.displayName, attendanceSummary); replyText(replyToken, geminiResponse); } else { replyText(replyToken, callGemini(text)); } } return ContentService.createTextOutput("OK"); } // ========================================== // ฟังก์ชัน Helper ต่างๆ // ========================================== function sendJsonResponse(successStatus, messageText) { const result = JSON.stringify({ success: successStatus, message: messageText }); return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON); } // ตรวจสอบการเช็คชื่อซ้ำ (ปรับปรุงเพื่อป้องกัน Bug จาก Object Date ของ Google Sheets) function hasCheckedInToday(userId, currentDate) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { let cellDate = data[i][0]; // ตรวจสอบและแปลงค่าให้เป็นข้อความฟอร์แมต dd/MM/yyyy ก่อนทำการเปรียบเทียบ if (cellDate instanceof Date) { cellDate = Utilities.formatDate(cellDate, "GMT+7", "dd/MM/yyyy"); } else if (cellDate) { cellDate = cellDate.toString(); } if (cellDate === currentDate && data[i][3] === userId) { return true; } } return false; } function calculateDistance(lat1, lon1, lat2, lon2) { const R = 6371e3; const dLat = (lat2 - lat1) * Math.PI / 180; const dLon = (lon2 - lon1) * Math.PI / 180; const a = Math.sin(dLat/2) * Math.sin(dLat/2) + Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2); const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a)); return R * c; } function saveDataToSheet(userId, displayName, status, source) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); const now = new Date(); const dateStr = Utilities.formatDate(now, "GMT+7", "dd/MM/yyyy"); const timeStr = Utilities.formatDate(now, "GMT+7", "HH:mm:ss"); sheet.appendRow([dateStr, timeStr, displayName, userId, status, source]); } // ดึงสรุปยอด (ปรับปรุงฟอร์แมตวันที่ในส่วนประวัติการเช็คชื่อย้อนหลัง) function getUserAttendanceSummary(userId) { const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); let present = 0, leave = 0, absent = 0, historyLogs = []; for (let i = 1; i < data.length; i++) { if (data[i][3] === userId) { if (data[i][4] === 'มาเรียน') present++; else if (data[i][4] === 'ลา') leave++; else if (data[i][4] === 'ขาด') absent++; let logDate = data[i][0]; if (logDate instanceof Date) { logDate = Utilities.formatDate(logDate, "GMT+7", "dd/MM/yyyy"); } historyLogs.push(`- วันที่ ${logDate}: ${data[i][4]}`); } } return { present, leave, absent, total: present+leave+absent, history: historyLogs.slice(-5).join('\n') }; } function getUserProfile(userId) { const res = UrlFetchApp.fetch(`https://api.line.me/v2/bot/profile/${userId}`, { headers: { 'Authorization': 'Bearer ' + LINE_ACCESS_TOKEN }, muteHttpExceptions: true }); return JSON.parse(res.getContentText()); } function replyText(replyToken, text) { UrlFetchApp.fetch('https://api.line.me/v2/bot/message/reply', { method: 'post', headers: { 'Content-Type': 'application/json', 'Authorization': 'Bearer ' + LINE_ACCESS_TOKEN }, payload: JSON.stringify({ replyToken: replyToken, messages: [{'type': 'text', 'text': text}] }) }); } function callGemini(text) { const res = UrlFetchApp.fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_API_KEY}`, { method: 'post', contentType: 'application/json', muteHttpExceptions: true, payload: JSON.stringify({ "contents": [{ "parts": [{"text": text}] }] }) }); const json = JSON.parse(res.getContentText()); return json.error ? "Error: " + json.error.message : json.candidates[0].content.parts[0].text; } function callGeminiWithAttendance(text, name, summary) { const instruction = `ข้อมูลเช็คชื่อคุณ ${name}: มา ${summary.present}, ลา ${summary.leave}, ขาด ${summary.absent}. ประวัติล่าสุด:\n${summary.history||"ไม่มี"}\nตอบกลับอย่างเป็นมิตร`; const res = UrlFetchApp.fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${GEMINI_API_KEY}`, { method: 'post', contentType: 'application/json', muteHttpExceptions: true, payload: JSON.stringify({ "contents": [{ "parts": [{"text": text}] }], "systemInstruction": { "parts": [{"text": instruction}] } }) }); const json = JSON.parse(res.getContentText()); return json.error ? "Error: " + json.error.message : json.candidates[0].content.parts[0].text; }
